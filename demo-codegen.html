TypeScript Service Class with Axios

import axios, { AxiosInstance } from 'axios';
import { ApiResponse, ExampleModel } from './interfaces';

export class ApiService {
    private client: AxiosInstance;

    constructor(baseURL: string, private getToken?: () => string | null) {
        this.client = axios.create({
            baseURL,
            timeout: 10000,
            headers: { 'Content-Type': 'application/json' }
        });
    }

    /**
     * Get all items
     */
    async getAll(): Promise<ExampleModel[]> {
        const response = await this.client.get<ExampleModel[]>('/api/examples');
        return response.data;
    }
}
